# Redis面试题 - 应用场景篇

## 1. Redis如何实现分布式锁？有哪些注意事项？

**解答：**

### 分布式锁的基本要求

```
1. 互斥性：同一时刻只能有一个客户端持有锁
2. 安全性：锁只能被持有者释放，不能被其他客户端释放
3. 死锁避免：客户端崩溃后，锁能够自动释放
4. 容错性：Redis集群部分节点故障，锁仍然可用（可选）
```

### 基础实现

#### 1. 基于SETNX的实现

**加锁：**
```bash
# 原子性设置锁
SET lock:order:123 owner_id NX EX 30

# NX - 只有当key不存在时才设置
# EX 30 - 设置30秒过期时间
```

**解锁：**
```bash
# 使用Lua脚本保证原子性
if redis.call("get", KEYS[1]) == ARGV[1] then
    return redis.call("del", KEYS[1])
else
    return 0
end
```

#### 2. Java完整实现

```java
@Component
public class RedisDistributedLock {
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    private static final String LOCK_PREFIX = "lock:";
    private static final long DEFAULT_EXPIRE_TIME = 30;
    
    // 解锁Lua脚本
    private static final String UNLOCK_SCRIPT = 
        "if redis.call('get', KEYS[1]) == ARGV[1] then " +
        "return redis.call('del', KEYS[1]) " +
        "else return 0 end";
    
    /**
     * 尝试获取锁
     * @param lockKey 锁的唯一标识
     * @param requestId 请求ID（用于释放锁时验证）
     * @param expireTime 过期时间（秒）
     * @return 是否获取成功
     */
    public boolean tryLock(String lockKey, String requestId, long expireTime) {
        String key = LOCK_PREFIX + lockKey;
        
        Boolean success = redisTemplate.opsForValue()
            .setIfAbsent(key, requestId, expireTime, TimeUnit.SECONDS);
        
        return Boolean.TRUE.equals(success);
    }
    
    /**
     * 尝试获取锁（带重试）
     */
    public boolean tryLockWithRetry(String lockKey, String requestId, 
            long expireTime, int retryTimes, long retryInterval) {
        for (int i = 0; i < retryTimes; i++) {
            if (tryLock(lockKey, requestId, expireTime)) {
                return true;
            }
            
            try {
                Thread.sleep(retryInterval);
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
                return false;
            }
        }
        return false;
    }
    
    /**
     * 释放锁
     * @param lockKey 锁的唯一标识
     * @param requestId 请求ID（必须和加锁时的requestId一致）
     * @return 是否释放成功
     */
    public boolean unlock(String lockKey, String requestId) {
        String key = LOCK_PREFIX + lockKey;
        
        try {
            // 执行Lua脚本原子性释放锁
            Long result = redisTemplate.execute(
                new DefaultRedisScript<>(UNLOCK_SCRIPT, Long.class),
                Collections.singletonList(key),
                requestId
            );
            
            return result != null && result > 0;
        } catch (Exception e) {
            log.error("释放锁失败", e);
            return false;
        }
    }
    
    /**
     * 使用try-finally安全释放锁
     */
    public <T> T executeWithLock(String lockKey, long expireTime, 
            Supplier<T> supplier) {
        String requestId = UUID.randomUUID().toString();
        boolean locked = false;
        
        try {
            locked = tryLock(lockKey, requestId, expireTime);
            if (!locked) {
                throw new RuntimeException("获取锁失败");
            }
            
            return supplier.get();
        } finally {
            if (locked) {
                unlock(lockKey, requestId);
            }
        }
    }
}
```

```java
@Service
public class RedissonLockService {
    
    @Autowired
    private RedissonClient redissonClient;
    
    /**
     * 可重入锁
     */
    public void reentrantLock(String lockKey) {
        RLock lock = redissonClient.getLock(lockKey);
        try {
            // 尝试获取锁，最多等待10秒，30秒后自动释放
            boolean locked = lock.tryLock(10, 30, TimeUnit.SECONDS);
            if (locked) {
                try {
                    // 业务逻辑
                    
                    // 可重入：同一线程可以再次获取锁
                    lock.lock();
                    try {
                        // 嵌套业务逻辑
                    } finally {
                        lock.unlock();
                    }
                    
                } finally {
                    // 确保释放锁
                    if (lock.isHeldByCurrentThread()) {
                        lock.unlock();
                    }
                }
            }
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
    }
    
    /**
     * 公平锁
     */
    public void fairLock(String lockKey) {
        // 公平锁：按请求锁的顺序获取锁
        RLock fairLock = redissonClient.getFairLock(lockKey);
        fairLock.lock();
        try {
            // 业务逻辑
        } finally {
            fairLock.unlock();
        }
    }
    
    /**
     * 读写锁
     */
    public void readWriteLock(String lockKey) {
        RReadWriteLock rwLock = redissonClient.getReadWriteLock(lockKey);
        
        // 读锁
        RLock readLock = rwLock.readLock();
        readLock.lock();
        try {
            // 读操作（多个线程可同时持有读锁）
        } finally {
            readLock.unlock();
        }
        
        // 写锁
        RLock writeLock = rwLock.writeLock();
        writeLock.lock();
        try {
            // 写操作（独占）
        } finally {
            writeLock.unlock();
        }
    }
    
    /**
     * 信号量（限流）
     */
    public void semaphore() {
        RSemaphore semaphore = redissonClient.getSemaphore("semaphore");
        // 设置许可证数量（并发数）
        semaphore.trySetPermits(10);
        
        try {
            // 获取许可证（阻塞直到有可用）
            semaphore.acquire();
            try {
                // 执行业务（最多10个并发）
            } finally {
                semaphore.release();
            }
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
    }
    
    /**
     * 联锁（同时获取多个锁）
     */
    public void multiLock() {
        RLock lock1 = redissonClient.getLock("lock1");
        RLock lock2 = redissonClient.getLock("lock2");
        RLock lock3 = redissonClient.getLock("lock3");
        
        // 联锁：同时获取所有锁
        RLock multiLock = redissonClient.getMultiLock(lock1, lock2, lock3);
        multiLock.lock();
        try {
            // 业务逻辑
        } finally {
            multiLock.unlock();
        }
    }
}
```

### 注意事项

#### 1. 锁的过期时间设置

```java
/**
 * 过期时间必须大于业务执行时间
 * 否则业务还没执行完，锁就过期了
 */
public void setExpireTime() {
    // 不好的做法：时间太短
    tryLock("order", requestId, 3);  // 3秒可能不够
    
    // 好的做法：时间充足
    tryLock("order", requestId, 30);  // 30秒通常足够
    
    // 更好的做法：使用Redisson自动续期
    redissonLock.lock();  // 不指定时间，启用看门狗
}
```

#### 2. 解锁时验证身份

```java
/**
 * 必须验证锁是否由当前客户端持有
 * 避免误删其他客户端的锁
 */
public void safeUnlock() {
    // 错误做法：直接删除
    redisTemplate.delete(lockKey);
    // 问题：可能删除了其他客户端获取的锁
    
    // 正确做法：使用Lua脚本验证身份
    String script = 
        "if redis.call('get', KEYS[1]) == ARGV[1] then " +
        "return redis.call('del', KEYS[1]) " +
        "else return 0 end";
    
    redisTemplate.execute(script, Collections.singletonList(lockKey), requestId);
}
```

#### 3. 避免死锁

```java
/**
 * 必须使用try-finally确保锁释放
 */
public void avoidDeadlock() {
    boolean locked = false;
    try {
        locked = tryLock(lockKey, requestId, 30);
        if (locked) {
            // 业务逻辑
        }
    } catch (Exception e) {
        log.error("业务异常", e);
    } finally {
        // 确保释放锁
        if (locked) {
            unlock(lockKey, requestId);
        }
    }
}
```

#### 4. 锁的可重入性

```java
/**
 * 同一个线程应该能够多次获取同一把锁
 */
public void reentrant() {
    // 基础实现不支持可重入
    // 第二次tryLock会失败（key已存在）
    
    // Redisson支持可重入
    RLock lock = redissonClient.getLock("lock");
    lock.lock();
    try {
        lock.lock();  // 同一线程可以再次获取
        try {
            // 业务逻辑
        } finally {
            lock.unlock();  // 释放第二次获取的锁
        }
    } finally {
        lock.unlock();  // 释放第一次获取的锁
    }
}
```

## 2. Redis如何实现延时队列？

**解答：**

### 方案对比

| 方案 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| Sorted Set | 精确到毫秒，支持删除 | 需要轮询 | 精确的延时任务 |
| List + BRPOPLPUSH | 简单，阻塞读取 | 只能实现固定间隔 | 简单的任务队列 |
| Stream | 功能强大，持久化 | 复杂度高 | 复杂的消息队列 |
| 过期回调 | 无需轮询 | 不保证及时性 | 对时间要求不严格 |

### Sorted Set方案（推荐）

#### 原理

```
使用Sorted Set的score存储执行时间戳：
- member：任务ID或任务内容
- score：执行时间戳（毫秒）

消费流程：
1. 使用ZRANGEBYSCORE查询到期的任务（score <= 当前时间）
2. 使用ZREM删除任务（保证任务只被消费一次）
3. 使用Lua脚本保证查询和删除的原子性
```

#### 完整实现

```java
@Component
public class RedisDelayQueue {
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    private static final String DELAY_QUEUE_KEY = "delay:queue";
    
    /**
     * 添加延时任务
     * @param taskId 任务ID
     * @param taskData 任务数据
     * @param delayTime 延迟时间（毫秒）
     */
    public void addTask(String taskId, String taskData, long delayTime) {
        long executeTime = System.currentTimeMillis() + delayTime;
        String value = taskId + ":" + taskData;
        
        redisTemplate.opsForZSet().add(DELAY_QUEUE_KEY, value, executeTime);
    }
    
    /**
     * 批量获取到期任务（使用Lua脚本保证原子性）
     */
    public List<DelayTask> fetchDueTasks(int batchSize) {
        long now = System.currentTimeMillis();
        
        // Lua脚本：获取并删除到期任务
        String luaScript = 
            "local tasks = redis.call('zrangebyscore', KEYS[1], 0, ARGV[1], 'limit', 0, ARGV[2]); " +
            "if #tasks > 0 then " +
            "  redis.call('zremrangebyrank', KEYS[1], 0, #tasks - 1); " +
            "end; " +
            "return tasks;";
        
        DefaultRedisScript<List> script = new DefaultRedisScript<>();
        script.setScriptText(luaScript);
        script.setResultType(List.class);
        
        List<String> results = redisTemplate.execute(script,
            Collections.singletonList(DELAY_QUEUE_KEY),
            String.valueOf(now),
            String.valueOf(batchSize));
        
        if (results == null || results.isEmpty()) {
            return Collections.emptyList();
        }
        
        return results.stream()
            .map(this::parseTask)
            .filter(Objects::nonNull)
            .collect(Collectors.toList());
    }
    
    /**
     * 删除指定任务
     */
    public boolean removeTask(String taskId) {
        // 使用scan找到并删除（因为value包含taskData）
        Cursor<ZSetOperations.TypedTuple<String>> cursor = 
            redisTemplate.opsForZSet().scan(DELAY_QUEUE_KEY, 
                ScanOptions.scanOptions().match(taskId + ":*").count(100).build());
        
        while (cursor.hasNext()) {
            ZSetOperations.TypedTuple<String> tuple = cursor.next();
            if (tuple.getValue().startsWith(taskId + ":")) {
                redisTemplate.opsForZSet().remove(DELAY_QUEUE_KEY, tuple.getValue());
                return true;
            }
        }
        return false;
    }
    
    /**
     * 解析任务
     */
    private DelayTask parseTask(String value) {
        int index = value.indexOf(':');
        if (index < 0) return null;
        
        String taskId = value.substring(0, index);
        String taskData = value.substring(index + 1);
        
        return new DelayTask(taskId, taskData);
    }
    
    /**
     * 启动消费线程
     */
    @PostConstruct
    public void startConsumer() {
        ThreadPoolExecutor executor = new ThreadPoolExecutor(
            2, 4, 60, TimeUnit.SECONDS,
            new LinkedBlockingQueue<>(100),
            new ThreadFactoryBuilder().setNameFormat("delay-queue-%d").build()
        );
        
        // 启动多个消费线程
        for (int i = 0; i < 2; i++) {
            executor.submit(() -> {
                while (!Thread.currentThread().isInterrupted()) {
                    try {
                        List<DelayTask> tasks = fetchDueTasks(100);
                        if (tasks.isEmpty()) {
                            // 没有任务，休眠100ms
                            Thread.sleep(100);
                            continue;
                        }
                        
                        // 处理任务
                        for (DelayTask task : tasks) {
                            processTask(task);
                        }
                    } catch (Exception e) {
                        log.error("处理延时任务异常", e);
                        try {
                            Thread.sleep(1000);
                        } catch (InterruptedException ie) {
                            Thread.currentThread().interrupt();
                        }
                    }
                }
            });
        }
    }
    
    /**
     * 处理任务
     */
    private void processTask(DelayTask task) {
        try {
            log.info("处理延时任务: {}, 数据: {}", task.getTaskId(), task.getTaskData());
            // 根据taskId路由到对应的处理器
            TaskHandler handler = getHandler(task.getTaskId());
            if (handler != null) {
                handler.handle(task.getTaskData());
            }
        } catch (Exception e) {
            log.error("处理任务失败: {}", task.getTaskId(), e);
            // 失败重试逻辑
        }
    }
    
    private TaskHandler getHandler(String taskId) {
        // 根据taskId前缀路由
        if (taskId.startsWith("order:cancel:")) {
            return new OrderCancelHandler();
        } else if (taskId.startsWith("coupon:expire:")) {
            return new CouponExpireHandler();
        }
        return null;
    }
}
```

### List + BRPOPLPUSH方案

#### 原理

```
使用两个List实现延时队列：
- job_pool：存储任务数据
- delay_bucket：按时间分桶

但Redis原生的BRPOPLPUSH不支持延时，
需要配合定时任务或外部调度器使用

更简单的方式：使用固定间隔的阻塞读取
```

#### 实现

```java
@Component
public class SimpleDelayQueue {
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    /**
     * 添加任务（固定延迟，不支持自定义时间）
     */
    public void addTask(String taskData) {
        // 使用LPUSH入队
        redisTemplate.opsForList().leftPush("queue:pending", taskData);
    }
    
    /**
     * 消费任务（阻塞读取）
     */
    public void consumeTask() {
        while (true) {
            try {
                // BRPOP阻塞读取，超时时间10秒
                ListOperations<String, String> ops = redisTemplate.opsForList();
                String taskData = ops.rightPop("queue:pending", 10, TimeUnit.SECONDS);
                
                if (taskData != null) {
                    // 处理任务
                    processTask(taskData);
                }
            } catch (Exception e) {
                log.error("消费任务异常", e);
            }
        }
    }
    
    /**
     * 延时投递（使用定时任务重新入队）
     */
    @Scheduled(fixedDelay = 1000)  // 每秒检查
    public void retryFailedTasks() {
        // 检查失败任务队列，延迟重试
        String taskData = redisTemplate.opsForList()
            .rightPop("queue:failed");
        
        if (taskData != null) {
            // 延迟5秒后重新入队
            CompletableFuture.runAsync(() -> {
                try {
                    Thread.sleep(5000);
                    redisTemplate.opsForList().leftPush("queue:pending", taskData);
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                }
            });
        }
    }
}
```

### Stream方案（Redis 5.0+）

```java
@Component
public class RedisStreamDelayQueue {
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    private static final String STREAM_KEY = "delay:stream";
    private static final String CONSUMER_GROUP = "delay-group";
    
    /**
     * 添加延时消息
     */
    public void addMessage(String message, long delayMs) {
        long executeTime = System.currentTimeMillis() + delayMs;
        
        Map<String, String> content = new HashMap<>();
        content.put("data", message);
        content.put("executeTime", String.valueOf(executeTime));
        
        redisTemplate.opsForStream().add(STREAM_KEY, content);
    }
    
    /**
     * 创建消费者组
     */
    @PostConstruct
    public void createConsumerGroup() {
        try {
            redisTemplate.opsForStream().createGroup(STREAM_KEY, CONSUMER_GROUP);
        } catch (Exception e) {
            // 组已存在
        }
    }
    
    /**
     * 消费消息
     */
    public void consumeMessage() {
        while (true) {
            try {
                long now = System.currentTimeMillis();
                
                // 读取消息（阻塞）
                List<MapRecord<String, Object, Object>> messages = 
                    redisTemplate.opsForStream().read(
                        Consumer.from(CONSUMER_GROUP, "consumer-1"),
                        StreamReadOptions.empty().count(10),
                        StreamOffset.create(STREAM_KEY, ReadOffset.lastConsumed())
                    );
                
                if (messages == null || messages.isEmpty()) {
                    Thread.sleep(100);
                    continue;
                }
                
                for (MapRecord<String, Object, Object> message : messages) {
                    Map<Object, Object> value = message.getValue();
                    String data = (String) value.get("data");
                    long executeTime = Long.parseLong((String) value.get("executeTime"));
                    
                    if (executeTime <= now) {
                        // 时间到了，处理消息
                        processMessage(data);
                        
                        // ACK确认
                        redisTemplate.opsForStream()
                            .acknowledge(STREAM_KEY, CONSUMER_GROUP, message.getId());
                    } else {
                        // 时间还没到，重新投递（NACK）
                        // Stream不直接支持NACK，可以通过XCLAIM实现
                    }
                }
            } catch (Exception e) {
                log.error("消费消息异常", e);
            }
        }
    }
    
    private void processMessage(String data) {
        // 处理消息
    }
}
```

## 3. Redis如何实现限流？

**解答：**

### 限流算法

| 算法 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| 计数器 | 简单 | 临界问题 | 简单限流 |
| 滑动窗口 | 精确 | 内存占用大 | 精确限流 |
| 令牌桶 | 平滑 | 复杂 | 平滑限流 |
| 漏桶 | 平滑输出 | 突发流量处理差 | 恒定速率输出 |

### 计数器算法

#### 原理

```
使用String存储计数：
key: rate_limit:user:1000:202312011030（用户ID+分钟级时间）
value: 当前分钟内请求次数

流程：
1. INCR key（原子性增加）
2. 如果是第一次（返回1），设置过期时间60秒
3. 如果计数超过阈值，拒绝请求
```

#### 实现

```java
@Component
public class CounterRateLimiter {
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    /**
     * 固定窗口计数器限流
     * @param key 限流key
     * @param limit 限制次数
     * @param window 时间窗口（秒）
     * @return 是否允许通过
     */
    public boolean tryAcquire(String key, int limit, int window) {
        String rateKey = "rate:counter:" + key;
        
        Long count = redisTemplate.opsForValue().increment(rateKey);
        
        if (count == 1) {
            // 第一次访问，设置过期时间
            redisTemplate.expire(rateKey, window, TimeUnit.SECONDS);
        }
        
        return count <= limit;
    }
    
    /**
     * 按用户限流（分钟级）
     */
    public boolean tryAcquireByUser(Long userId, int limit) {
        String timeKey = LocalDateTime.now().format(DateTimeFormatter.ofPattern("yyyyMMddHHmm"));
        String key = userId + ":" + timeKey;
        return tryAcquire(key, limit, 60);
    }
    
    /**
     * 按接口限流（秒级）
     */
    public boolean tryAcquireByApi(String apiPath, int limit) {
        String timeKey = String.valueOf(System.currentTimeMillis() / 1000);
        String key = apiPath + ":" + timeKey;
        return tryAcquire(key, limit, 1);
    }
}
```

**问题：临界问题**
```
窗口切换时的流量突增：

时间线：
0-59秒：窗口1，允许100次请求
60-119秒：窗口2，允许100次请求

攻击方式：
55-59秒：发送100次请求
60-64秒：发送100次请求

在5秒内发送了200次请求，超过限流
```

### 滑动窗口算法

#### 原理

```
使用Sorted Set存储请求时间戳：
key: rate:slide:user:1000
score: 请求时间戳（毫秒）
member: 请求ID（或时间戳+序列号）

流程：
1. ZADD添加当前请求
2. ZREMRANGEBYSCORE删除窗口外的旧请求
3. ZCARD统计当前窗口内请求数
4. 如果超过阈值，删除刚添加的请求并拒绝
```

#### 实现

```java
@Component
public class SlidingWindowRateLimiter {
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    /**
     * 滑动窗口限流（使用Lua脚本保证原子性）
     */
    public boolean tryAcquire(String key, int limit, int windowMs) {
        String rateKey = "rate:slide:" + key;
        long now = System.currentTimeMillis();
        long windowStart = now - windowMs;
        String member = now + ":" + ThreadLocalRandom.current().nextInt(10000);
        
        // Lua脚本保证原子性
        String luaScript = 
            "redis.call('zremrangebyscore', KEYS[1], 0, ARGV[1]); " +
            "local current = redis.call('zcard', KEYS[1]); " +
            "if current < tonumber(ARGV[2]) then " +
            "  redis.call('zadd', KEYS[1], ARGV[3], ARGV[4]); " +
            "  redis.call('expire', KEYS[1], ARGV[5]); " +
            "  return 1; " +
            "else " +
            "  return 0; " +
            "end";
        
        DefaultRedisScript<Long> script = new DefaultRedisScript<>();
        script.setScriptText(luaScript);
        script.setResultType(Long.class);
        
        Long result = redisTemplate.execute(script,
            Collections.singletonList(rateKey),
            String.valueOf(windowStart),    // ARGV[1]: 窗口开始时间
            String.valueOf(limit),          // ARGV[2]: 限制次数
            String.valueOf(now),            // ARGV[3]: 当前时间
            member,                         // ARGV[4]: member
            String.valueOf(windowMs / 1000 + 1)  // ARGV[5]: 过期时间
        );
        
        return result != null && result == 1;
    }
    
    /**
     * 获取当前窗口内的请求数
     */
    public long getCurrentCount(String key, int windowMs) {
        String rateKey = "rate:slide:" + key;
        long now = System.currentTimeMillis();
        long windowStart = now - windowMs;
        
        // 清理过期数据
        redisTemplate.opsForZSet().removeRangeByScore(rateKey, 0, windowStart);
        
        // 获取当前计数
        Long count = redisTemplate.opsForZSet().zCard(rateKey);
        return count != null ? count : 0;
    }
}

// 使用示例

            return ResponseEntity.status(429)
                .body("请求过于频繁，请稍后再试");
        }
        
        // 执行业务逻辑
        return ResponseEntity.ok(data);
    }
}
```

### 令牌桶算法

#### 原理

```
使用Hash存储令牌桶状态：
key: rate:token:user:1000
field: 
  - last_time: 上次更新时间
  - tokens: 当前令牌数

流程：
1. 计算时间差，补充令牌
2. 如果有可用令牌，消费一个令牌
3. 如果没有令牌，拒绝请求
```

#### 实现

```java
@Component
public class TokenBucketRateLimiter {
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    /**
     * 令牌桶限流（使用Lua脚本）
     * @param key 限流key
     * @param capacity 桶容量
     * @param rate 每秒产生令牌数
     * @return 是否获取到令牌
     */
    public boolean tryAcquire(String key, int capacity, double rate) {
        String bucketKey = "rate:token:" + key;
        long now = System.currentTimeMillis();
        
        String luaScript = 
            "local bucket = redis.call('hmget', KEYS[1], 'tokens', 'last_time'); " +
            "local tokens = tonumber(bucket[1]) or " + capacity + "; " +
            "local last_time = tonumber(bucket[2]) or " + now + "; " +
            "local now = " + now + "; " +
            "local rate = " + rate + "; " +
            "local capacity = " + capacity + "; " +
            "local delta = math.max(0, now - last_time) / 1000; " +
            "tokens = math.min(capacity, tokens + delta * rate); " +
            "local allowed = 0; " +
            "if tokens >= 1 then " +
            "  tokens = tokens - 1; " +
            "  allowed = 1; " +
            "end; " +
            "redis.call('hmset', KEYS[1], 'tokens', tokens, 'last_time', now); " +
            "redis.call('expire', KEYS[1], 60); " +
            "return allowed;";
        
        DefaultRedisScript<Long> script = new DefaultRedisScript<>();
        script.setScriptText(luaScript);
        script.setResultType(Long.class);
        
        Long result = redisTemplate.execute(script, 
            Collections.singletonList(bucketKey));
        
        return result != null && result == 1;
    }
    
    /**
     * 获取当前令牌数
     */
    public double getTokens(String key, int capacity, double rate) {
        String bucketKey = "rate:token:" + key;
        
        List<String> values = redisTemplate.opsForHash()
            .multiGet(bucketKey, Arrays.asList("tokens", "last_time"));
        
        if (values.get(0) == null) {
            return capacity;
        }
        
        double tokens = Double.parseDouble(values.get(0));
        long lastTime = Long.parseLong(values.get(1));
        long now = System.currentTimeMillis();
        
        double delta = Math.max(0, now - lastTime) / 1000.0;
        return Math.min(capacity, tokens + delta * rate);
    }
}
```

```java
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    private static final String LEADERBOARD_KEY = "leaderboard:global";

     * 更新玩家分数

     * 增加玩家分数
     */

    }
    
    /**
     * 获取前N名
     */
    public List<PlayerRank> getTopN(int n) {
        Set<ZSetOperations.TypedTuple<String>> topN = 
            redisTemplate.opsForZSet().reverseRangeWithScores(LEADERBOARD_KEY, 0, n - 1);
        
        if (topN == null || topN.isEmpty()) {
            return Collections.emptyList();

        if (rank == null) {
            return null;
        }
        
        // 获取分数

            playerRank.setPlayerId(tuple.getValue());
            playerRank.setScore(tuple.getScore());
            playerRank.setRank(currentRank++);
            playerRank.setCurrentPlayer(tuple.getValue().equals(playerId));

        }
        
        return result;
```

## 5. Redis如何实现消息队列？

**解答：**

### 方案对比

| 方案 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| List | 简单，支持阻塞读取 | 无ACK机制 | 简单队列 |
| Pub/Sub | 支持多播 | 无持久化，消息丢失 | 实时推送 |
| Stream | 功能完善，持久化 | 复杂度高 | 生产级消息队列 |

### List实现消息队列

#### 基本原理

```bash
# 生产者入队
LPUSH queue:messages "message1"
LPUSH queue:messages "message2"

# 消费者出队（阻塞读取）
BRPOP queue:messages 0

# BRPOP参数说明：
# 第一个参数：队列key
# 第二个参数：超时时间（0表示永久阻塞）
# 返回：key和value
```

#### Java实现

```java
@Component
public class ListMessageQueue {
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    private static final String QUEUE_KEY = "queue:default";
    
    /**
     * 发送消息
     */
    public void sendMessage(String message) {
        redisTemplate.opsForList().leftPush(QUEUE_KEY, message);
    }
    
    /**
     * 批量发送消息
     */
    public void sendMessages(List<String> messages) {
        redisTemplate.executePipelined((RedisCallback<Object>) connection -> {
            for (String message : messages) {
                connection.lPush(QUEUE_KEY.getBytes(), message.getBytes());
            }
            return null;
        });
    }
    
    /**
     * 消费消息（阻塞）
     */
    public String consumeMessage(long timeoutSeconds) {
        ListOperations<String, String> ops = redisTemplate.opsForList();
        // BRPOP返回的是[key, value]列表
        List<String> result = ops.rightPop(QUEUE_KEY, timeoutSeconds, TimeUnit.SECONDS);
        
        if (result == null || result.isEmpty()) {
            return null;
        }
        
        // result[0] = key, result[1] = value
        return result.get(1);
    }
    
    /**
     * 启动消费者
     */
    @PostConstruct
    public void startConsumer() {
        ThreadPoolExecutor executor = new ThreadPoolExecutor(
            2, 4, 60, TimeUnit.SECONDS,
            new LinkedBlockingQueue<>(100)
        );
        
        for (int i = 0; i < 2; i++) {
            executor.submit(() -> {
                while (!Thread.currentThread().isInterrupted()) {
                    try {
                        String message = consumeMessage(10);
                        if (message != null) {
                            processMessage(message);
                        }
                    } catch (Exception e) {
                        log.error("消费消息异常", e);
                    }
                }
            });
        }
    }
    
    private void processMessage(String message) {
        log.info("处理消息: {}", message);
        // 业务逻辑
    }
    
    /**
     * 获取队列长度
     */
    public long getQueueLength() {
        Long len = redisTemplate.opsForList().size(QUEUE_KEY);
        return len != null ? len : 0;
    }
}
```

**问题：无ACK机制，消息可能丢失**
```
消费者取出消息后崩溃：
1. BRPOP取出消息
2. 消费者处理消息时崩溃
3. 消息丢失

解决方案：
1. 先备份再处理（RPOP + LPUSH到备份队列）
2. 使用Lua脚本实现原子操作
```

```java
/**
 * 带ACK的消息队列（使用备份队列）
 */
@Component
public class ReliableListQueue {
    
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    private static final String QUEUE_KEY = "queue:reliable";
    private static final String BACKUP_KEY = "queue:backup";
    
    /**
     * 可靠消费（原子性迁移到备份队列）
     */
    public String consumeReliable(long timeoutSeconds) {
        // Lua脚本：原子性从队列迁移到备份队列
        String luaScript = 
            "local msg = redis.call('brpop', KEYS[1], ARGV[1]); " +
            "if msg then " +
            "  redis.call('lpush', KEYS[2], msg[2]); " +
            "  redis.call('expire', KEYS[2], 300); " +  // 备份5分钟
            "  return msg[2]; " +
            "end; " +
            "return nil;";
        
        DefaultRedisScript<String> script = new DefaultRedisScript<>();
        script.setScriptText(luaScript);
        script.setResultType(String.class);
        
        String message = redisTemplate.execute(script,
            Arrays.asList(QUEUE_KEY, BACKUP_KEY),
            String.valueOf(timeoutSeconds));
        
        return message;
    }
    
    /**
     * ACK确认（从备份队列删除）
     */
    public void ackMessage(String message) {
        redisTemplate.opsForList().remove(BACKUP_KEY, 0, message);
    }
    
    /**
     * 失败重试（从备份队列重新入队）
     */
    public void retryFailedMessages() {
        String message;
        while ((message = redisTemplate.opsForList().rightPop(BACKUP_KEY)) != null) {
            redisTemplate.opsForList().leftPush(QUEUE_KEY, message);
        }
    }
    
    /**
     * 定时重试失败消息
     */
    @Scheduled(fixedDelay = 60000)  // 每分钟检查
    public void scheduledRetry() {
        retryFailedMessages();
    }
}
```

**问题：无持久化，消息可能丢失**
```
Pub/Sub的缺点：
1. 无持久化，订阅者不在线时消息丢失
2. 消费者断开后不会保留消息

- 实时通知
- 在线状态广播
- 不保证必达的消息
```

### Stream实现消息队列（推荐）

#### 基本原理

```bash
# 添加消息（自动分配ID）
XADD mystream * field1 value1 field2 value2

# 创建消费者组
XGROUP CREATE mystream mygroup $ MKSTREAM

# 消费者组消费
XREADGROUP GROUP mygroup consumer1 COUNT 1 STREAMS mystream >

# ACK确认
XACK mystream mygroup 1526569495631-0

# 查看未确认消息
XPENDING mystream mygroup
```

```java
    @Autowired
    private StringRedisTemplate redisTemplate;
    
    private final ExecutorService executorService = Executors.newFixedThreadPool(4);
    
    /**
     * 启动消费者
     */
    @PostConstruct
    public void startConsumer() {
        for (int i = 0; i < 4; i++) {
            String consumerName = "consumer-" + i;
            executorService.submit(() -> consumeMessages(consumerName));
        }
    }
    
    /**
     * 消费消息
     */
    public void consumeMessages(String consumerName) {
        while (!Thread.currentThread().isInterrupted()) {
            try {
                // 读取消息（阻塞）
                List<MapRecord<String, Object, Object>> messages = 
                    redisTemplate.opsForStream().read(
                        Consumer.from(RedisStreamConfig.CONSUMER_GROUP, consumerName),
                        StreamReadOptions.empty().count(10).block(Duration.ofSeconds(5)),
                        StreamOffset.create(RedisStreamConfig.STREAM_KEY, ReadOffset.lastConsumed())
                    );
                
                if (messages == null || messages.isEmpty()) {
                    continue;
                }
                
                for (MapRecord<String, Object, Object> message : messages) {
                    try {
                        processMessage(message);
                        
                        // ACK确认
                        redisTemplate.opsForStream().acknowledge(
                            RedisStreamConfig.STREAM_KEY,
                            RedisStreamConfig.CONSUMER_GROUP,
                            message.getId()
                        );
                    } catch (Exception e) {
                        log.error("处理消息失败: {}", message.getId(), e);
                        // 消息会保留在PEL（Pending Entries List）中，等待重试
                    }
                }
            } catch (Exception e) {
                log.error("消费消息异常", e);
            }
        }
    }

        Map<Object, Object> value = message.getValue();
        
        String orderId = (String) value.get("orderId");
        String action = (String) value.get("action");

            processOrderPay(orderId, value);

            processOrderCancel(orderId, value);
        }
    }

    }
    
    private void processOrderCancel(String orderId, Map<Object, Object> data) {
        // 处理订单取消
    }
    
    /**
     * 处理未确认消息（死信队列）
     */
    @Scheduled(fixedDelay = 300000)  // 每5分钟检查
    public void processPendingMessages() {
        // 获取未确认消息
        PendingMessagesSummary summary = redisTemplate.opsForStream()
            .pending(RedisStreamConfig.STREAM_KEY, RedisStreamConfig.CONSUMER_GROUP);
        
        if (summary.getTotalPendingMessages() == 0) {
            return;
        }
        
        // 获取详细未确认消息
        PendingMessages pendingMessages = redisTemplate.opsForStream()
            .pending(RedisStreamConfig.STREAM_KEY, 
                    RedisStreamConfig.CONSUMER_GROUP,

                    Range.closed("-", "+"),

                    pending.getId(), elapsed.toMinutes());

            }
        }
```

### 消息队列对比

| 特性 | List | Pub/Sub | Stream |
|------|------|---------|--------|
| 持久化 | 是 | 否 | 是 |
| ACK机制 | 无（需自己实现） | 无 | 有 |
| 消费者组 | 无 | 无 | 有 |
| 消息回溯 | 是 | 否 | 是 |
| 阻塞读取 | 是 | 是 | 是 |
| 复杂度 | 低 | 低 | 高 |
| 适用场景 | 简单队列 | 实时推送 | 生产级消息队列 |

**推荐选择：**
- **简单场景**：List + BRPOP
- **实时通知**：Pub/Sub
- **生产环境**：Stream（功能最完善）
